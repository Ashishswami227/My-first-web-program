app.post('/api/posts/:id/comment', async (req, res) => { try { const { username, text } = req.body; const post = await Post.findById(req.params.id); if (!post) return res.status(404).json({ message: "Post not found." }); post.comments.push({ username, text }); await post.save(); res.json(post); } catch (err) { res.status(500).json({ error: err.message }); } });
