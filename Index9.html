const mongoose = require('mongoose'); const CommentSchema = new mongoose.Schema({ username: String, text: String, createdAt: { type: Date, default: Date.now } }); const PostSchema = new mongoose.Schema({ title: { type: String, required: true }, content: { type: String, required: true }, author: { type: String, required: true }, comments: [CommentSchema] }, { timestamps: true }); module.exports = mongoose.model('Post', PostSchema);
