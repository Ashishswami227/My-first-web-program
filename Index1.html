const express = require('express'); const mongoose = require('mongoose'); const app = express(); const PORT = process.env.PORT || 3000; app.use(express.json()); app.use(express.static('.')); mongoose.connect('mongodb://localhost:27017/portfolio', { useNewUrlParser: true, useUnifiedTopology: true }).then(() => console.log("MongoDB Connected")) .catch(err => console.log(err)); app.get('/api/projects', async (req, res) => { res.json([{title: "Portfolio", description: "Full stack project"}]); }); app.listen(PORT, () => console.log(`Server running on port ${PORT}`));
